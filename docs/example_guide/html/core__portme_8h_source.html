133/* Configuration : MULTITHREAD
134 Define for parallel execution
135
136 Valid values :
137 1 - only one context (default).
138 N>1 - will execute N copies in parallel.
139
140 Note :
141 If this flag is defined to more then 1, an implementation for launching parallel contexts must be defined.
142
143 Two sample implementations are provided. Use <USE_PTHREAD> or <USE_FORK> to enable them.
144
145 It is valid to have a different implementation of <core_start_parallel> and <core_end_parallel> in <core_portme.c>,
146 to fit a particular architecture.
147*/
148#ifndef MULTITHREAD

150#define USE_PTHREAD 0
151#define USE_FORK 0
152#define USE_SOCKET 0
153#endif
