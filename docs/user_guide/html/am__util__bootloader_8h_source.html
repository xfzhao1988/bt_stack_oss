66//*****************************************************************************
67//
68// Structure to keep track of boot image information.
69// In the flash, the structure contains a 4 byte CRC for integrity
70// verification. It needs to be ensured that the size of this structure is
71// not more than AM_HAL_FLASH_PAGE_SIZE bytes
72//
73//*****************************************************************************
74typedef struct
75{
76 // Starting address where the image was linked to run.
77 uint32_t *pui32LinkAddress;
78
79 // Length of the executable image in bytes.
80 uint32_t ui32NumBytes;
81
82 // CRC-32 Value for the full image.
83 uint32_t ui32CRC;
84
85 // Override GPIO number. (Can be used to force a new image load)
86 uint32_t ui32OverrideGPIO;
87
88 // Polarity for the override pin.
89 uint32_t ui32OverridePolarity;
90
91 // Stack pointer location.
92 uint32_t *pui32StackPointer;
93
94 // Reset vector location.
95 uint32_t *pui32ResetVector;
96
97 // Protection status of image in flash
98 uint32_t bEncrypted;
99
100 // CRC-32 value of this structure
101 uint32_t ui32Checksum;
102}
