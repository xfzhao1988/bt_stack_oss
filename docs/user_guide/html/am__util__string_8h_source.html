70#define AM_CATTR_DIGIT 0x08

75//
76// Set MINIMIZE_CATTR_TABLE to 1 to configure for minimal CATTR table size,
77// (256 instead of 512 bytes) but at a cost of slightly larger code size.
78// However, setting this option also provides an additional level of checking
79// of the argument; if the argument is not a uint8_t, the functions are
80// guaranteed to return 0.
